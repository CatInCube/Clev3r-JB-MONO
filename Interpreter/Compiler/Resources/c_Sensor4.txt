// ------------------------------------------ EV3 MODULE: SENSOR1 ----------------------------------------------

DATA32 s4out1
DATA32 s4out2
DATA32 s4out3

init
{
	MOVE32_32 0 s4out1
	MOVE32_32 0 s4out2
	MOVE32_32 0 s4out3
}

inline SENSOR4.RAW1    // F
{
	INPUT_READEXT 0 3 0 -1 18 1 s4out1
	MOVE32_F s4out1 :0
}

inline SENSOR4.RAW3    // FFFV
{
	INPUT_READEXT 0 3 0 -1 18 3 s4out1 s4out2 s4out3
	MOVE32_F s4out1 :0
	MOVE32_F s4out2 :1
	MOVE32_F s4out3 :2
}
